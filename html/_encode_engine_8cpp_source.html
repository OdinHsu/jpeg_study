96 void EncodeEngine::Run()
97 {
98  unsigned int flags=0;
99 
100  m_first = true;
101 
102  DBGTRACE(DBG_THREAD,"EncodeEngine: Thread start.\n");
103  SetEvent(sync_event);
104 
105  while( 1 ) {
106  // Wait for events
107  pthread_mutex_lock(&m_engine.mutex);
108  while(!m_engine.flag) pthread_cond_wait(&m_engine.cond,&m_engine.mutex);
109  // keep flags
110  flags = m_engine.flag & EVENT_ALL;
111  m_engine.flag = 0;
112  pthread_mutex_unlock(&m_engine.mutex);
113 
114  // Process events
115  while( flags ) {
116  if(flags & EVENT_REMOVE) {
117  DBGTRACE(DBG_ENGINE,"EncodeEngine: receive EVENT_REMOVE\n");
118  flags ^= EVENT_REMOVE;
119 
120  DBGTRACE(DBG_THREAD,"EncodeEngine: Thread terminate.\n");
121  return;
122  };
123  if(flags & EVENT_ENCODE) {
124  DBGTRACE(DBG_ENGINE,"EncodeEngine: receive EVENT_ENCODE\n");
125  flags ^= EVENT_ENCODE;
126 
127  Analysis();
128  SetEvent(sync_event);
129  };
130  };
131  };
