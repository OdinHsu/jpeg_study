184 bool BlockEncoder::CheckFrameBufferBlock(int x,int y,BYTE* framebuffer)
185 {
186 
187  int height = BLOCK_HEIGHT;
188  unsigned int idx;
189 
190  UINT32* p_src;
191  UINT32* p_dest;
192 
193  if( (y == m_vBlocks-1) && (m_vLastBlackValid != 0) ) height = m_vLastBlackValid;
194 
195  idx = y*m_hBlocks + x;
196 
197  x = x*BLOCK_WIDTH; // convert from block to pixel
198  y = y*BLOCK_HEIGHT; // convert from block number to line
199 
200  int buf_idx = (1 ^ p_compress->buff_idx[idx]) & 0x01;
201 
202  int dest = idx*2+buf_idx;
203  int src = idx*2+(1 ^ buf_idx);
204 
205  p_src = (UINT32*) &framebuffer[(y*m_pitch+x)*RAW_DEPTH]; // block buffer
206  p_dest = (UINT32*) &p_analysis->pframebuffer[dest*BLOCK_SIZE]; // linear buffer
207 
208  for(int i=0;i<height;i++) {
209  pRGBRow[i] = (BYTE*)p_src;
210  pYRow[i] = (BYTE*)p_dest;
211  pVRow[i] = pYRow[i] + COLOR_SIZE;
212  pURow[i] = pVRow[i] + COLOR_SIZE;
213  p_src += m_pitch;
214  p_dest += BLOCK_WIDTH/4;
215  };
216 
217  jsimd_extbgrx_ycc_convert( BLOCK_WIDTH, pRGBRow, component, 0, height );
218 
219  p_src = (UINT32*) &p_analysis->pframebuffer[idx*2*BLOCK_SIZE];
220  p_dest = (UINT32*) &p_analysis->pframebuffer[(idx*2+1)*BLOCK_SIZE];
221 
222  if(memcmp(p_src,p_dest,BLOCK_SIZE) == 0) {
223  bool new_block = (p_compress->buff_idx[idx] != p_analysis->buff_idx[idx]);
224  p_analysis->buff_idx[idx] = p_compress->buff_idx[idx];

268 unsigned int BlockEncoder::get_bitstream(BYTE* bitstream)
269 {
270  int index,len,i;
271  int quality __attribute__((unused));
272  int* pp;
273  int block_length;
274  BYTE* pstream = bitstream;
275  blockheader_t* pblock_header=NULL;
276 
277  if(p_compress->encode_count == 0) return 0;
278 
279  for(i=0;i<p_compress->encode_count;i++) {
280  _UNPACK(index, quality, p_compress->encode_list[i]);
281 
282  // Tag header
283  pblock_header = (blockheader_t*)pstream;
284  pstream += sizeof(DWORD);
285 
286  // write header
287  *(DWORD*)pblock_header = p_compress->pheader[index];
288 
289  // get length from header
290  block_length = (int)pblock_header->len + 1;
291 
292  // write payload
293  memcpy(pstream,p_compress->pencode_buffer+(index*BLOCK_SIZE),block_length*4);
294  pstream += block_length*4;
295  }
296  pblock_header->end = 1; // last header
297  len = (int)(pstream - bitstream);
298  pp = (int*)pstream;
299 
300  // Padding 0xFFFFD9FF to 128 bytes alignment ( AXI burst length )
301  do{
302  *pp++ = 0xFFFFD9FF; // end of JPEG and end of frame
303  len += 4;
304  }while(len%128);
305  // end of alignment
306  return len;
307 };

401 int EncodeUnit::EncodeBlocks()
402 {
403  int index,quality,uncompress_data;
404  blockheader_t* pblock_header=NULL;
405  int len = 0;
406  int jpeg_block_size = 0;
407  int count = block_end-block_start;
408 
409  for(int i=block_start;i<block_end;i++,count--) {
410  // get block index and quality from encode list
411  _UNPACK(index, quality, m_pScreen->encode_list[i]);
412 
413  if(quality == QUALITY_CLONE) {
414  pblock_header = (blockheader_t*) &m_pScreen->pheader[index];
415  len += (pblock_header->len+2)*4;
416  continue;
417  }
418 
419  m_pScreen->pheader[index] = 0xD0000000; // ID
420 
421  pblock_header = (blockheader_t*)&m_pScreen->pheader[index];
422  pblock_header->block_index = index;
423 
424  if( !m_first ) {

426  if( (uncompress_data + len) > data_size ) {
427  quality = QUALITY_MEDIAN;
428  }
429  }
430

436  };
437
