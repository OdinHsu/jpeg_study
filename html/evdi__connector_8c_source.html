30 static int evdi_get_modes(struct drm_connector *connector)
31 {
32  struct evdi_device *evdi = connector->dev->dev_private;
33  struct edid *edid = NULL;
34  int ret = 0;
35 
36  edid = (struct edid *)evdi_painter_get_edid_copy(evdi);
37 
38  if (!edid) {
39 #if KERNEL_VERSION(4, 19, 0) <= LINUX_VERSION_CODE || defined(EL8)
40  drm_connector_update_edid_property(connector, NULL);
41 #else
42  drm_mode_connector_update_edid_property(connector, NULL);
43 #endif
44  return 0;
45  }
46 
47 #if KERNEL_VERSION(4, 19, 0) <= LINUX_VERSION_CODE || defined(EL8)
48  ret = drm_connector_update_edid_property(connector, edid);
49 #else
50  ret = drm_mode_connector_update_edid_property(connector, edid);
51 #endif
52 
53  if (ret) {
54  EVDI_ERROR("Failed to set edid property! error: %d", ret);
55  goto err;
56  }
57 
58  ret = drm_add_edid_modes(connector, edid);
59  EVDI_INFO("(card%d) Edid property set", evdi->dev_index);
60 err:
61  kfree(edid);
62  return ret;
63 }

65 static enum drm_mode_status evdi_mode_valid(struct drm_connector *connector,
66  struct drm_display_mode *mode)
67 {
68  struct evdi_device *evdi = connector->dev->dev_private;
69  uint32_t area_limit = mode->hdisplay * mode->vdisplay;
70  uint32_t mode_limit = area_limit * drm_mode_vrefresh(mode);
71 
72  if (evdi->pixel_per_second_limit == 0)
73  return MODE_OK;
74 
75  if (area_limit > evdi->pixel_area_limit ||
76  mode_limit > evdi->pixel_per_second_limit) {
77  EVDI_WARN("(card%d) Mode %dx%d@%d rejected\n",
78  evdi->dev_index,
79  mode->hdisplay,
80  mode->vdisplay,
81  drm_mode_vrefresh(mode));
82  return MODE_BAD;
83  }
84 
85  return MODE_OK;
86 }
