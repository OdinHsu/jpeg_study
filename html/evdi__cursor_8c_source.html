118 void evdi_cursor_set(struct evdi_cursor *cursor,
119  struct evdi_gem_object *obj,
120  uint32_t width, uint32_t height,
121  int32_t hot_x, int32_t hot_y,
122  uint32_t pixel_format, uint32_t stride)

126  evdi_cursor_lock(cursor);

135  cursor->enabled = obj != NULL;
136  cursor->width = width;
137  cursor->height = height;
138  cursor->hot_x = hot_x;
139  cursor->hot_y = hot_y;
140  cursor->pixel_format = pixel_format;
141  cursor->stride = stride;
142  evdi_cursor_set_gem(cursor, obj);
143 
144  evdi_cursor_unlock(cursor);
145 }

155 static inline uint32_t blend_component(uint32_t pixel,
156  uint32_t blend,
157  uint32_t alpha)
158 {
159  uint32_t pre_blend = (pixel * (255 - alpha) + blend * alpha);
160 
161  return (pre_blend + ((pre_blend + 1) << 8)) >> 16;
162 }
163 
164 static inline uint32_t blend_alpha(const uint32_t pixel_val32,
165  uint32_t blend_val32)
166 {
167  uint32_t alpha = (blend_val32 >> 24);
168 
169  return blend_component(pixel_val32 & 0xff,
170  blend_val32 & 0xff, alpha) |
171  blend_component((pixel_val32 & 0xff00) >> 8,
172  (blend_val32 & 0xff00) >> 8, alpha) << 8 |
173  blend_component((pixel_val32 & 0xff0000) >> 16,
174  (blend_val32 & 0xff0000) >> 16, alpha) << 16;
175 }

187 int evdi_cursor_compose_and_copy(struct evdi_cursor *cursor,
188  struct evdi_framebuffer *efb,
189  char __user *buffer,
190  int buf_byte_stride)
191 {
192  int x, y;
193  struct drm_framebuffer *fb = &efb->base;
194  const int h_cursor_w = cursor->width >> 1;
195  const int h_cursor_h = cursor->height >> 1;
196  uint32_t *cursor_buffer = NULL;
197  uint32_t bytespp = 0;
198 
199  if (!cursor->enabled)
200  return 0;
201 
202  if (!cursor->obj)
203  return -EINVAL;
204 
205  if (!cursor->obj->vmapping)
206  return -EINVAL;
207 
208  bytespp = evdi_fb_get_bpp(cursor->pixel_format);
209  bytespp = DIV_ROUND_UP(bytespp, 8);
210  if (bytespp != 4) {
211  EVDI_ERROR("Unsupported cursor format bpp=%u\n", bytespp);
212  return -EINVAL;
213  }
214 
215  if (cursor->width * cursor->height * bytespp >
216  cursor->obj->base.size){
217  EVDI_ERROR("Wrong cursor size\n");
218  return -EINVAL;
219  }
220 
221  cursor_buffer = (uint32_t *)cursor->obj->vmapping;
222 
223  for (y = -h_cursor_h; y < h_cursor_h; ++y) {
224  for (x = -h_cursor_w; x < h_cursor_w; ++x) {
225  uint32_t curs_val;
226  int *fbsrc;
227  int fb_value;
228  int cmd_offset;
229  int cursor_pix;
230  int const mouse_pix_x = cursor->x + x + h_cursor_w;
231  int const mouse_pix_y = cursor->y + y + h_cursor_h;
232  bool const is_pix_sane =
233  mouse_pix_x >= 0 &&
234  mouse_pix_y >= 0 &&
235  mouse_pix_x < (int)fb->width &&
236  mouse_pix_y < (int)fb->height;
237 
238  if (!is_pix_sane)
239  continue;
240 
241  cursor_pix = h_cursor_w+x +
242  (h_cursor_h+y)*cursor->width;
243  curs_val = le32_to_cpu(cursor_buffer[cursor_pix]);
244  fbsrc = (int *)(efb->obj->vmapping + fb->offsets[0]);
245  fb_value = *(fbsrc + ((fb->pitches[0]>>2) *
246  mouse_pix_y + mouse_pix_x));
247  cmd_offset = (buf_byte_stride * mouse_pix_y) +
248  (mouse_pix_x * bytespp);
249  if (evdi_cursor_compose_pixel(buffer,
250  curs_val,
251  fb_value,
252  cmd_offset)) {
253  EVDI_ERROR("Failed to compose cursor pixel\n");
254  return -EFAULT;
255  }
256  }
257  }
258 
259  return 0;
260 }
