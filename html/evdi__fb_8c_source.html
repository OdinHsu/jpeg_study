42 struct drm_clip_rect evdi_framebuffer_sanitize_rect(
43  const struct evdi_framebuffer *fb,
44  const struct drm_clip_rect *dirty_rect)
45 {
46  struct drm_clip_rect rect = *dirty_rect;
47 
48  if (rect.x1 > rect.x2) {
49  unsigned short tmp = rect.x2;
50 
51  EVDI_WARN("Wrong clip rect: x1 > x2\n");
52  rect.x2 = rect.x1;
53  rect.x1 = tmp;
54  }
55 
56  if (rect.y1 > rect.y2) {
57  unsigned short tmp = rect.y2;
58 
59  EVDI_WARN("Wrong clip rect: y1 > y2\n");
60  rect.y2 = rect.y1;
61  rect.y1 = tmp;
62  }
63 
64 
65  if (rect.x1 > fb->base.width) {
66  EVDI_DEBUG("Wrong clip rect: x1 > fb.width\n");
67  rect.x1 = fb->base.width;
68  }
69 
70  if (rect.y1 > fb->base.height) {
71  EVDI_DEBUG("Wrong clip rect: y1 > fb.height\n");
72  rect.y1 = fb->base.height;
73  }
74 
75  if (rect.x2 > fb->base.width) {
76  EVDI_DEBUG("Wrong clip rect: x2 > fb.width\n");
77  rect.x2 = fb->base.width;
78  }
79 
80  if (rect.y2 > fb->base.height) {
81  EVDI_DEBUG("Wrong clip rect: y2 > fb.height\n");
82  rect.y2 = fb->base.height;
83  }
84 
85  return rect;
86 }

108 static int evdi_fb_mmap(struct fb_info *info, struct vm_area_struct *vma)
109 {
110  unsigned long start = vma->vm_start;
111  unsigned long size = vma->vm_end - vma->vm_start;
112  unsigned long offset = vma->vm_pgoff << PAGE_SHIFT;
113  unsigned long page, pos;
114 
115  if (vma->vm_pgoff > (~0UL >> PAGE_SHIFT))
116  return -EINVAL;
117 
118  if (offset > info->fix.smem_len ||
119  size > info->fix.smem_len - offset)
120  return -EINVAL;
121 
122  pos = (unsigned long)info->fix.smem_start + offset;
123 
124  pr_notice("mmap() framebuffer addr:%lu size:%lu\n", pos, size);
125 
126  while (size > 0) {
127  page = vmalloc_to_pfn((void *)pos);
128  if (remap_pfn_range(vma, start, page, PAGE_SIZE, PAGE_SHARED))
129  return -EAGAIN;
130 
131  start += PAGE_SIZE;
132  pos += PAGE_SIZE;
133  if (size > PAGE_SIZE)
134  size -= PAGE_SIZE;
135  else
136  size = 0;
137  }
138 
139  return 0;
140 }

230 static int evdi_user_framebuffer_dirty(
231  struct drm_framebuffer *fb,
232  __maybe_unused struct drm_file *file_priv,
233  __always_unused unsigned int flags,
234  __always_unused unsigned int color,
235  __always_unused struct drm_clip_rect *clips,
236  __always_unused unsigned int num_clips)
237 {
238  struct evdi_framebuffer *efb = to_evdi_fb(fb);
239  struct drm_device *dev = efb->base.dev;
240  struct evdi_device *evdi = dev->dev_private;
241 
242  struct drm_modeset_acquire_ctx ctx;
243  struct drm_atomic_state *state;
244  struct drm_plane *plane;
245  int ret = 0;
246  unsigned int i;
247 
248  EVDI_CHECKPT();
249 
250  drm_modeset_acquire_init(&ctx,
251  /*
252  * When called from ioctl, we are interruptable,
253  * but not when called internally (ie. defio worker)
254  */
255  file_priv ? DRM_MODESET_ACQUIRE_INTERRUPTIBLE : 0);
256 
257  state = drm_atomic_state_alloc(fb->dev);
258  if (!state) {
259  ret = -ENOMEM;
260  goto out;
261  }
262  state->acquire_ctx = &ctx;
263 
264  for (i = 0; i < num_clips; ++i)
265  evdi_painter_mark_dirty(evdi, &clips[i]);
266 
267 retry:
268 
269  drm_for_each_plane(plane, fb->dev) {
270  struct drm_plane_state *plane_state;
271 
272  if (plane->state->fb != fb)
273  continue;
274 
275  /*
276  * Even if it says 'get state' this function will create and
277  * initialize state if it does not exists. We use this property
278  * to force create state.
279  */
280  plane_state = drm_atomic_get_plane_state(state, plane);
281  if (IS_ERR(plane_state)) {
282  ret = PTR_ERR(plane_state);
283  goto out;
284  }
285  }
286 
287  ret = drm_atomic_commit(state);
288 
289 out:
290  if (ret == -EDEADLK) {
291  drm_atomic_state_clear(state);
292  ret = drm_modeset_backoff(&ctx);
293  if (!ret)
294  goto retry;
295  }
296 
297  if (state)
298  drm_atomic_state_put(state);
299 
300  drm_modeset_drop_locks(&ctx);
301  drm_modeset_acquire_fini(&ctx);
302 
303  return ret;
304 }

353 static int evdifb_create(struct drm_fb_helper *helper,
354  struct drm_fb_helper_surface_size *sizes)
355 {
356  struct evdi_fbdev *efbdev = (struct evdi_fbdev *)helper;
357  struct drm_device *dev = efbdev->helper.dev;
358  struct fb_info *info;
359  struct device *device = dev->dev;
360  struct drm_framebuffer *fb;
361  struct drm_mode_fb_cmd2 mode_cmd;
362  struct evdi_gem_object *obj;
363  uint32_t size;
364  int ret = 0;
365 
366  if (sizes->surface_bpp == 24) {
367  sizes->surface_bpp = 32;
368  } else if (sizes->surface_bpp != 32) {
369  EVDI_ERROR("Not supported pixel format (bpp=%d)\n",
370  sizes->surface_bpp);
371  return -EINVAL;
372  }
373 
374  mode_cmd.width = sizes->surface_width;
375  mode_cmd.height = sizes->surface_height;
376  mode_cmd.pitches[0] = mode_cmd.width * ((sizes->surface_bpp + 7) / 8);
377 
378  mode_cmd.pixel_format = drm_mode_legacy_fb_format(sizes->surface_bpp,
379  sizes->surface_depth);
380 
381  size = mode_cmd.pitches[0] * mode_cmd.height;
382  size = ALIGN(size, PAGE_SIZE);
383 
384  obj = evdi_gem_alloc_object(dev, size);
385  if (!obj)
386  goto out;
387 
388  ret = evdi_gem_vmap(obj);
389  if (ret) {
390  DRM_ERROR("failed to vmap fb\n");
391  goto out_gfree;
392  }
393 
394  info = framebuffer_alloc(0, device);
395  if (!info) {
396  ret = -ENOMEM;
397  goto out_gfree;
398  }
399  info->par = efbdev;
400 
401  ret = evdi_framebuffer_init(dev, &efbdev->efb, &mode_cmd, obj);
402  if (ret)
403  goto out_gfree;
404 
405  fb = &efbdev->efb.base;
406 
407  efbdev->helper.fb = fb;
408  efbdev->helper.fbdev = info;
409 
410  strcpy(info->fix.id, "evdidrmfb");
411 
412  info->screen_base = efbdev->efb.obj->vmapping;
413  info->fix.smem_len = size;
414  info->fix.smem_start = (unsigned long)efbdev->efb.obj->vmapping;
415 
416 #if KERNEL_VERSION(4, 20, 0) <= LINUX_VERSION_CODE || defined(EL8)
417  info->flags = FBINFO_DEFAULT;
418 #else
419  info->flags = FBINFO_DEFAULT | FBINFO_CAN_FORCE_OUTPUT;
420 #endif
421 
422  efbdev->fb_ops = evdifb_ops;
423  info->fbops = &efbdev->fb_ops;
424 
425 #if KERNEL_VERSION(5, 2, 0) <= LINUX_VERSION_CODE || defined(EL8)
426  drm_fb_helper_fill_info(info, &efbdev->helper, sizes);
427 #else
428  drm_fb_helper_fill_fix(info, fb->pitches[0], fb->format->depth);
429  drm_fb_helper_fill_var(info, &efbdev->helper, sizes->fb_width,
430  sizes->fb_height);
431 #endif
432 
433  ret = fb_alloc_cmap(&info->cmap, 256, 0);
434  if (ret) {
435  ret = -ENOMEM;
436  goto out_gfree;
437  }
438 
439  DRM_DEBUG_KMS("allocated %dx%d vmal %p\n",
440  fb->width, fb->height, efbdev->efb.obj->vmapping);
441 
442  return ret;
443  out_gfree:
444 #if KERNEL_VERSION(5, 9, 0) <= LINUX_VERSION_CODE || defined(EL8)
445  drm_gem_object_put(&efbdev->efb.obj->base);
446 #else
447  drm_gem_object_put_unlocked(&efbdev->efb.obj->base);
448 #endif
449  out:
450  return ret;
451 }

562 struct drm_framebuffer *evdi_fb_user_fb_create(
563  struct drm_device *dev,
564  struct drm_file *file,
565  const struct drm_mode_fb_cmd2 *mode_cmd)
566 {
567  struct drm_gem_object *obj;
568  struct evdi_framebuffer *efb;
569  int ret;
570  uint32_t size;
571  int bpp = evdi_fb_get_bpp(mode_cmd->pixel_format);
572 
573  if (bpp != 32) {
574  EVDI_ERROR("Unsupported bpp (%d)\n", bpp);
575  return ERR_PTR(-EINVAL);
576  }
577 
578  obj = drm_gem_object_lookup(file, mode_cmd->handles[0]);
579  if (obj == NULL)
580  return ERR_PTR(-ENOENT);
581 
582  size = mode_cmd->offsets[0] + mode_cmd->pitches[0] * mode_cmd->height;
583  size = ALIGN(size, PAGE_SIZE);
584 
585  if (size > obj->size) {
586  DRM_ERROR("object size not sufficient for fb %d %zu %u %d %d\n",
587  size, obj->size, mode_cmd->offsets[0],
588  mode_cmd->pitches[0], mode_cmd->height);
589  goto err_no_mem;
590  }
591 
592  efb = kzalloc(sizeof(*efb), GFP_KERNEL);
593  if (efb == NULL)
594  goto err_no_mem;
595  efb->base.obj[0] = obj;
596 
597  ret = evdi_framebuffer_init(dev, efb, mode_cmd, to_evdi_bo(obj));
598  if (ret)
599  goto err_inval;
600  return &efb->base;
601 
602  err_no_mem:
603  drm_gem_object_put(obj);
604  return ERR_PTR(-ENOMEM);
605  err_inval:
606  kfree(efb);
607  drm_gem_object_put(obj);
608  return ERR_PTR(-EINVAL);
609 }
