55 static int compat_evdi_connect(struct file *file,
56  unsigned int __always_unused cmd,
57  unsigned long arg)
58 {
59  struct drm_evdi_connect32 req32;
60  struct drm_evdi_connect __user *request;
61 
62  if (copy_from_user(&req32, (void __user *)arg, sizeof(req32)))
63  return -EFAULT;
64 
65 #if KERNEL_VERSION(5, 0, 0) <= LINUX_VERSION_CODE && KERNEL_VERSION(5, 14, 0) >= LINUX_VERSION_CODE || defined(EL8)
66  request = compat_alloc_user_space(sizeof(*request));
67 #else
68  request = kmalloc(sizeof(*request), GFP_USER);
69 #endif
70 #if KERNEL_VERSION(5, 0, 0) <= LINUX_VERSION_CODE || defined(EL8)
71  if (!access_ok(request, sizeof(*request))
72 #else
73  if (!access_ok(VERIFY_WRITE, request, sizeof(*request))
74 #endif
75  || __put_user(req32.connected, &request->connected)
76  || __put_user(req32.dev_index, &request->dev_index)
77  || __put_user((void __user *)(unsigned long)req32.edid_ptr32,
78  &request->edid)
79  || __put_user(req32.edid_length, &request->edid_length)
80  || __put_user(req32.pixel_area_limit, &request->pixel_area_limit)
81  || __put_user(req32.pixel_per_second_limit, &request->pixel_per_second_limit))
82  return -EFAULT;
83 
84  return drm_ioctl(file, DRM_IOCTL_EVDI_CONNECT,
85  (unsigned long)request);
86 }

137 long evdi_compat_ioctl(struct file *filp, unsigned int cmd, unsigned long arg)
138 {
139  unsigned int nr = DRM_IOCTL_NR(cmd);
140  drm_ioctl_compat_t *fn = NULL;
141  int ret;
142 
143  if (nr < DRM_COMMAND_BASE || nr >= DRM_COMMAND_END)
144  return drm_compat_ioctl(filp, cmd, arg);
145 
146  if (nr < DRM_COMMAND_BASE + ARRAY_SIZE(evdi_compat_ioctls))
147  fn = evdi_compat_ioctls[nr - DRM_COMMAND_BASE];
148 
149  if (fn != NULL)
150  ret = (*fn) (filp, cmd, arg);
151  else
152  ret = drm_ioctl(filp, cmd, arg);
153 
154  return ret;
155 }
