87 static int drm_is_master(int fd)
88 {
89  /* Detect master by attempting something that requires master.
90  *
91  * Authenticating magic tokens requires master and 0 is an
92  * internal kernel detail which we could use. Attempting this on
93  * a master fd would fail therefore fail with EINVAL because 0
94  * is invalid.
95  *
96  * A non-master fd will fail with EACCES, as the kernel checks
97  * for master before attempting to do anything else.
98  *
99  * Since we don't want to leak implementation details, use
100  * EACCES.
101  */
102  return drm_auth_magic(fd, 0) != -EACCES;
103 }

148 /*

155 static int removeFrameBuffer(evdi_handle context, int const *id)
156 {
157  struct evdi_frame_buffer_node *current = NULL;
158  struct evdi_frame_buffer_node *next = NULL;
159  struct evdi_frame_buffer_node **prev = NULL;
160  int removedCount = 0;
161 
162  current = context->frameBuffersListHead;
163  prev = &context->frameBuffersListHead;
164  while (current) {
165  next = current->next;
166

168  free(current);

172  prev = &current->next;
173  }
174 
175  current = next;
176  }
177

309 static int device_has_master(const char *device_file_path)
310 {
311  pid_t myself = getpid();
312  DIR *proc_dir;
313  struct dirent *proc_entry;
314  int result = 0;
315 
316  proc_dir = opendir("/proc");
317  if (proc_dir == NULL)
318  return 0;
319 
320  while ((proc_entry = readdir(proc_dir)) != NULL) {
321  char *d_name = proc_entry->d_name;
322 
323  if (d_name[0] < '0'
324  || d_name[0] > '9'
325  || myself == atoi(d_name)) {
326  continue;
327  }
328 
329  if (process_opened_files(d_name, device_file_path)) {
330  result = 1;
331  break;
332  }
333 
334  if (process_opened_device(d_name, device_file_path)) {
335  result = 1;
336  break;
337  }
338  }
339 
340  closedir(proc_dir);
341  return result;
342 }

361 {
362  int fd = 0;
363  int err = 0;
364 
365  fd = open(device_path, O_RDWR);
366  if (fd < 0)
367  return -1;

374 
375  if (err < 0) {
376  evdi_log("Drop master on %s failed, err: %s",

383  evdi_log("Drop master on %s failed, err: %s",
384  device_path, strerror(errno));
385  close(fd);
386  return -1;
387  }
388 
389  evdi_log("Opened %s as slave drm device", device_path);
390  return fd;

471 static bool is_correct_parent_device(const char *dirname, const char *parent_device)
472 {
473  char link_path[PATH_MAX];
474 
475  snprintf(link_path, PATH_MAX - 7, "%s/device", dirname);
476 
477  if (parent_device == NULL)
478  return access(link_path, F_OK) != 0;
479 
480  char link_resolution[PATH_MAX];
481  const ssize_t link_resolution_len = readlink(link_path, link_resolution, PATH_MAX);
482 
483  if (link_resolution_len == -1 || link_resolution_len == PATH_MAX)
484  return false;
485 
486  link_resolution[link_resolution_len] = '\0';
487  char *parent_device_token = strrchr(link_resolution, '/');
488 
489  if (strlen(parent_device) < 2)
490  return false;
491 
492  parent_device_token++;
493  size_t len = strlen(parent_device_token);
494 
495  bool is_same_device = strlen(parent_device) == len && strncmp(parent_device_token, parent_device, len) == 0;
496 
497  return is_same_device;
498 }

900 static void evdi_handle_event(evdi_handle handle,
901  struct evdi_event_context *evtctx,
902  struct drm_event *e)
903 {
904  switch (e->type) {
905  case DRM_EVDI_EVENT_UPDATE_READY:
906  if (evtctx->update_ready_handler)
907  evtctx->update_ready_handler(handle->bufferToUpdate,
908  evtctx->user_data);
909  break;
910 
911  case DRM_EVDI_EVENT_DPMS:
912  if (evtctx->dpms_handler) {
913  struct drm_evdi_event_dpms *event =
914  (struct drm_evdi_event_dpms *) e;
915 
916  evtctx->dpms_handler(event->mode,
917  evtctx->user_data);
918  }
919  break;
920 
921  case DRM_EVDI_EVENT_MODE_CHANGED:
922  if (evtctx->mode_changed_handler) {
923  struct drm_evdi_event_mode_changed *event =
924  (struct drm_evdi_event_mode_changed *) e;
925 
926  evtctx->mode_changed_handler(to_evdi_mode(event),
927  evtctx->user_data);
928  }
929  break;
930 
931  case DRM_EVDI_EVENT_CRTC_STATE:
932  if (evtctx->crtc_state_handler) {
933  struct drm_evdi_event_crtc_state *event =
934  (struct drm_evdi_event_crtc_state *) e;
935 
936  evtctx->crtc_state_handler(event->state,
937  evtctx->user_data);
938  }
939  break;
940 
941  case DRM_EVDI_EVENT_CURSOR_SET:
942  if (evtctx->cursor_set_handler) {
943  struct drm_evdi_event_cursor_set *event =
944  (struct drm_evdi_event_cursor_set *) e;
945  struct evdi_cursor_set cursor_set = to_evdi_cursor_set(handle, event);
946 
947  if (cursor_set.enabled && cursor_set.buffer == NULL)
948  evdi_log("Error: Cursor buffer is null!");
949  else
950  evtctx->cursor_set_handler(cursor_set,
951  evtctx->user_data);
952  }
953  break;
954 
955  case DRM_EVDI_EVENT_CURSOR_MOVE:
956  if (evtctx->cursor_move_handler) {
957  struct drm_evdi_event_cursor_move *event =
958  (struct drm_evdi_event_cursor_move *) e;
959 
960  evtctx->cursor_move_handler(to_evdi_cursor_move(event),
961  evtctx->user_data);
962  }
963  break;
964 
965  case DRM_EVDI_EVENT_DDCCI_DATA:
966  if (evtctx->ddcci_data_handler) {
967  struct drm_evdi_event_ddcci_data *event =
968  (struct drm_evdi_event_ddcci_data *) e;
969 
970  evtctx->ddcci_data_handler(to_evdi_ddcci_data(event),
971  evtctx->user_data);
972  }
973  break;
974 
975  default:
976  evdi_log("Warning: Unhandled event");
977  }
978 }

980 void evdi_handle_events(evdi_handle handle, struct evdi_event_context *evtctx)
981 {
982  char buffer[1024];
983  int i = 0;
984 
985  int bytesRead = read(handle->fd, buffer, sizeof(buffer));
986
