108 static int evdi_crtc_cursor_set(struct drm_crtc *crtc,
109  struct drm_file *file,
110  uint32_t handle,
111  uint32_t width,
112  uint32_t height,
113  int32_t hot_x,
114  int32_t hot_y)
115 {
116  struct drm_device *dev = crtc->dev;
117  struct evdi_device *evdi = dev->dev_private;
118  struct drm_gem_object *obj = NULL;
119  struct evdi_gem_object *eobj = NULL;
120  /*
121  * evdi_crtc_cursor_set is callback function using
122  * deprecated cursor entry point.
123  * There is no info about underlaying pixel format.
124  * Hence we are assuming that it is in ARGB 32bpp format.
125  * This format it the only one supported in cursor composition
126  * function.
127  * This format is also enforced during framebuffer creation.
128  *
129  * Proper format will be available when driver start support
130  * universal planes for cursor.
131  */
132  uint32_t format = DRM_FORMAT_ARGB8888;
133  uint32_t stride = 4 * width;
134 
135  EVDI_CHECKPT();
136  if (handle) {
137  mutex_lock(&dev->struct_mutex);
138  obj = drm_gem_object_lookup(file, handle);
139  if (obj)
140  eobj = to_evdi_bo(obj);
141  else
142  EVDI_ERROR("Failed to lookup gem object.\n");
143  mutex_unlock(&dev->struct_mutex);
144  }
145 
146  evdi_cursor_set(evdi->cursor,
147  eobj, width, height, hot_x, hot_y,
148  format, stride);
149 #if KERNEL_VERSION(5, 9, 0) <= LINUX_VERSION_CODE || defined(EL8)
150  drm_gem_object_put(obj);
151 #else
152  drm_gem_object_put_unlocked(obj);
153 #endif
154 
155  /*
156  * For now we don't care whether the application wanted the mouse set,
157  * or not.
158  */
159  if (evdi->cursor_events_enabled)
160  evdi_painter_send_cursor_set(evdi->painter, evdi->cursor);
161  else
162  evdi_mark_full_screen_dirty(evdi);
163  return 0;
164 }

222 static void evdi_plane_atomic_update(struct drm_plane *plane,
223 #if KERNEL_VERSION(5, 13, 0) <= LINUX_VERSION_CODE
224  struct drm_atomic_state *atom_state
225 #else
226  struct drm_plane_state *old_state
227 #endif
228  )
229 {
230 #if KERNEL_VERSION(5, 13, 0) <= LINUX_VERSION_CODE
231  struct drm_plane_state *old_state = drm_atomic_get_old_plane_state(atom_state, plane);
232 #else
233 #endif
234  struct drm_plane_state *state;
235  struct evdi_device *evdi;
236  struct evdi_painter *painter;
237  struct drm_crtc *crtc;
238 
239 #if KERNEL_VERSION(5, 0, 0) <= LINUX_VERSION_CODE || defined(EL8)
240  struct drm_atomic_helper_damage_iter iter;
241  struct drm_rect rect;
242  struct drm_clip_rect clip_rect;
243 #endif
244 
245  if (!plane || !plane->state) {
246  EVDI_WARN("Plane state is null\n");
247  return;
248  }
249 
250  if (!plane->dev || !plane->dev->dev_private) {
251  EVDI_WARN("Plane device is null\n");
252  return;
253  }
254 
255  state = plane->state;
256  evdi = plane->dev->dev_private;
257  painter = evdi->painter;
258  crtc = state->crtc;
259 
260  if (!old_state->crtc && state->crtc)
261  evdi_painter_dpms_notify(evdi, DRM_MODE_DPMS_ON);
262  else if (old_state->crtc && !state->crtc)
263  evdi_painter_dpms_notify(evdi, DRM_MODE_DPMS_OFF);
264 
265  if (state->fb) {
266  struct drm_framebuffer *fb = state->fb;
267  struct drm_framebuffer *old_fb = old_state->fb;
268  struct evdi_framebuffer *efb = to_evdi_fb(fb);
269 
270  const struct drm_clip_rect fullscreen_rect = {
271  0, 0, fb->width, fb->height
272  };
273 
274  if (!old_fb && crtc)
275  evdi_painter_force_full_modeset(painter);
276 
277  if (old_fb &&
278  fb->format && old_fb->format &&
279  fb->format->format != old_fb->format->format)
280  evdi_painter_force_full_modeset(painter);
281 
282  if (fb != old_fb ||
283  evdi_painter_needs_full_modeset(painter)) {
284 
285  evdi_painter_set_scanout_buffer(painter, efb);
286 
287 #if KERNEL_VERSION(5, 0, 0) <= LINUX_VERSION_CODE || defined(EL8)
288  state->visible = true;
289  state->src.x1 = 0;
290  state->src.y1 = 0;
291  state->src.x2 = fb->width << 16;
292  state->src.y2 = fb->height << 16;
293 
294  drm_atomic_helper_damage_iter_init(&iter, old_state, state);
295  while (drm_atomic_helper_damage_iter_next(&iter, &rect)) {
296  clip_rect.x1 = rect.x1;
297  clip_rect.y1 = rect.y1;
298  clip_rect.x2 = rect.x2;
299  clip_rect.y2 = rect.y2;
300  evdi_painter_mark_dirty(evdi, &clip_rect);
301  }
302 #endif
303 
304  };
305 
306  if (evdi_painter_get_num_dirts(painter) == 0)
307  evdi_painter_mark_dirty(evdi, &fullscreen_rect);
308  }
309 }

320 static void evdi_cursor_atomic_update(struct drm_plane *plane,
321 #if KERNEL_VERSION(5, 13, 0) <= LINUX_VERSION_CODE
322  struct drm_atomic_state *atom_state
323 #else
324  struct drm_plane_state *old_state
325 #endif
326  )
327 {
328 #if KERNEL_VERSION(5, 13, 0) <= LINUX_VERSION_CODE
329  struct drm_plane_state *old_state = drm_atomic_get_old_plane_state(atom_state, plane);
330 #else
331 #endif
332  if (plane && plane->state && plane->dev && plane->dev->dev_private) {
333  struct drm_plane_state *state = plane->state;
334  struct evdi_device *evdi = plane->dev->dev_private;
335  struct drm_framebuffer *fb = state->fb;
336  struct evdi_framebuffer *efb = to_evdi_fb(fb);
337 
338  struct drm_clip_rect old_rect;
339  struct drm_clip_rect rect;
340  bool cursor_changed = false;
341  bool cursor_position_changed = false;
342  int32_t cursor_position_x = 0;
343  int32_t cursor_position_y = 0;
344 
345  mutex_lock(&plane->dev->struct_mutex);
346 
347  evdi_cursor_position(evdi->cursor, &cursor_position_x,
348  &cursor_position_y);
349  evdi_cursor_move(evdi->cursor, state->crtc_x, state->crtc_y);
350  cursor_position_changed = cursor_position_x != state->crtc_x ||
351  cursor_position_y != state->crtc_y;
352 
353  if (fb != old_state->fb) {
354  if (fb != NULL) {
355  uint32_t stride = 4 * fb->width;
356 
357  evdi_cursor_set(evdi->cursor,
358  efb->obj,
359  fb->width,
360  fb->height,
361  0,
362  0,
363  fb->format->format,
364  stride);
365  }
366 
367  evdi_cursor_enable(evdi->cursor, fb != NULL);
368  cursor_changed = true;
369  }
370 
371  mutex_unlock(&plane->dev->struct_mutex);
372  if (!evdi->cursor_events_enabled) {
373  if (fb != NULL) {
374  if (efb->obj->allow_sw_cursor_rect_updates) {
375  evdi_cursor_atomic_get_rect(&old_rect, old_state);
376  evdi_cursor_atomic_get_rect(&rect, state);
377 
378  evdi_painter_mark_dirty(evdi, &old_rect);
379  } else {
380  rect = evdi_painter_framebuffer_size(evdi->painter);
381  }
382  evdi_painter_mark_dirty(evdi, &rect);
383  }
384  return;
385  }
386 
387  if (cursor_changed)
388  evdi_painter_send_cursor_set(evdi->painter,
389  evdi->cursor);
390  if (cursor_position_changed)
391  evdi_painter_send_cursor_move(evdi->painter,
392  evdi->cursor);
393  }
394 }
