122 static void merge_dirty_rects(struct drm_clip_rect *rects, int *count)
123 {
124  int a, b;
125 
126  for (a = 0; a < *count - 1; ++a) {
127  for (b = a + 1; b < *count;) {
128  /* collapse to bounding rect if it is fewer pixels */
129  const int area_a = rect_area(&rects[a]);
130  const int area_b = rect_area(&rects[b]);
131  struct drm_clip_rect bounding_rect = rects[a];
132 
133  expand_rect(&bounding_rect, &rects[b]);
134 
135  if (rect_area(&bounding_rect) <= area_a + area_b) {
136  rects[a] = bounding_rect;
137  rects[b] = rects[*count - 1];
138  /* repass */
139  b = a + 1;
140  --*count;
141  } else {
142  ++b;
143  }
144  }
145  }
146 }

160 static int copy_primary_pixels(struct evdi_framebuffer *efb,
161  char __user *buffer,
162  int buf_byte_stride,
163  int num_rects, struct drm_clip_rect *rects,
164  int const max_x,
165  int const max_y)
166 {
167  struct drm_framebuffer *fb = &efb->base;
168  struct drm_clip_rect *r;
169 
170  EVDI_CHECKPT();
171 
172  for (r = rects; r != rects + num_rects; ++r) {
173  const int byte_offset = r->x1 * 4;
174  const int byte_span = (r->x2 - r->x1) * 4;
175  const int src_offset = fb->offsets[0] +
176  fb->pitches[0] * r->y1 + byte_offset;
177  const char *src = (char *)efb->obj->vmapping + src_offset;
178  const int dst_offset = buf_byte_stride * r->y1 + byte_offset;
179  char __user *dst = buffer + dst_offset;
180  int y = r->y2 - r->y1;
181 
182  /* rect size may correspond to previous resolution */
183  if (max_x < r->x2 || max_y < r->y2) {
184  EVDI_WARN("Rect size beyond expected dimensions\n");
185  return -EFAULT;
186  }
187 
188  EVDI_VERBOSE("copy rect %d,%d-%d,%d\n", r->x1, r->y1, r->x2,
189  r->y2);
190 
191  for (; y > 0; --y) {
192  if (copy_to_user(dst, src, byte_span))
193  return -EFAULT;
194 
195  src += fb->pitches[0];
196  dst += buf_byte_stride;
197  }
198  }
199 
200  return 0;
201 }

289 static bool evdi_painter_flush_pending_events(struct evdi_painter *painter)
290 {
291  unsigned long flags;
292  struct drm_pending_event *event_to_be_sent = NULL;
293  struct list_head *list = NULL;
294  bool has_space = false;
295  bool flushed_all = false;
296 
297  spin_lock_irqsave(&painter->drm_device->event_lock, flags);
298 
299  list = &painter->pending_events;
300  while ((event_to_be_sent = list_first_entry_or_null(
301  list, struct drm_pending_event, link))) {
302  has_space = drm_event_reserve_init_locked(painter->drm_device,
303  painter->drm_filp, event_to_be_sent,
304  event_to_be_sent->event) == 0;
305  if (has_space) {
306  list_del_init(&event_to_be_sent->link);
307  drm_send_event_locked(painter->drm_device,
308  event_to_be_sent);
309  } else
310  break;
311  }
312 
313  flushed_all = list_empty(&painter->pending_events);
314  spin_unlock_irqrestore(&painter->drm_device->event_lock, flags);
315 
316  return flushed_all;
317 }

319 static void evdi_painter_send_event(struct evdi_painter *painter,
320  struct drm_pending_event *event)
321 {
322  if (!event) {
323  EVDI_ERROR("Null drm event!");
324  return;
325  }
326 
327  if (!painter->drm_filp) {
328  EVDI_VERBOSE("Painter is not connected!");
329  drm_event_cancel_free(painter->drm_device, event);
330  return;
331  }
332 
333  if (!painter->drm_device) {
334  EVDI_WARN("Painter is not connected to drm device!");
335  drm_event_cancel_free(painter->drm_device, event);
336  return;
337  }
338 
339  if (!painter->is_connected) {
340  EVDI_WARN("Painter is not connected!");
341  drm_event_cancel_free(painter->drm_device, event);
342  return;
343  }
344 
345  evdi_painter_add_event_to_pending_list(painter, event);
346  if (delayed_work_pending(&painter->send_events_work))
347  return;
348 
349  if (evdi_painter_flush_pending_events(painter))
350  return;
351 
352  schedule_delayed_work(&painter->send_events_work, msecs_to_jiffies(5));
353 }

1009  int ret;

1013

1015  EVDI_ERROR("Unknown command mode\n");

1017  }

1036  err = -EAGAIN;

1066  painter->vblank = NULL;

1086  EVDI_DEBUG("Invalid buffer dimension\n");
1087  err = -EINVAL;
1088  goto err_fb;
1089  }
1090

1092  cmd->num_rects * sizeof(cmd->rects[0]))) {
1093  err = -EFAULT;
1094  goto err_fb;
1095  }

1102  err = -EFAULT;
1103  goto err_fb;
1104  }
1105  }
1106 
1107  err = copy_primary_pixels(efb,
1108  cmd->buffer,
1109  cmd->buf_byte_stride,
1110  cmd->num_rects,
1111  dirty_rects,
1112  cmd->buf_width,
1113  cmd->buf_height);
1114  if (err == 0 && !evdi->cursor_events_enabled)
1115  copy_cursor_pixels(efb,
1116  cmd->buffer,
1117  cmd->buf_byte_stride,
1118  evdi->cursor);
1119 
1120  if (import_attach)
1121  dma_buf_end_cpu_access(import_attach->dmabuf,
1122  DMA_FROM_DEVICE);
1123 
1124 err_fb:
1125  evdi_send_vblank(crtc, vblank);
1126 
1127  drm_framebuffer_put(&efb->base);

1131 err_painter:

1134 }
