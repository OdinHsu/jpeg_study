59 int evdi_platform_device_probe(struct platform_device *pdev)
60 {
61  struct drm_device *dev;
62  struct evdi_platform_device_data *data;
63 
64 #if KERNEL_VERSION(5, 9, 0) <= LINUX_VERSION_CODE || defined(EL8)
65 #if IS_ENABLED(CONFIG_IOMMU_API) && defined(CONFIG_INTEL_IOMMU)
66  struct dev_iommu iommu;
67 #endif
68 #endif
69  EVDI_CHECKPT();
70 
71  data = kzalloc(sizeof(struct evdi_platform_device_data), GFP_KERNEL);
72  if (!data)
73  return -ENOMEM;
74 /* Intel-IOMMU workaround: platform-bus unsupported, force ID-mapping */
75 #if IS_ENABLED(CONFIG_IOMMU_API) && defined(CONFIG_INTEL_IOMMU)
76 #if KERNEL_VERSION(5, 9, 0) <= LINUX_VERSION_CODE || defined(EL8)
77  memset(&iommu, 0, sizeof(iommu));
78  iommu.priv = (void *)-1;
79  pdev->dev.iommu = &iommu;
80 #else
81 #define INTEL_IOMMU_DUMMY_DOMAIN ((void *)-1)
82  pdev->dev.archdata.iommu = INTEL_IOMMU_DUMMY_DOMAIN;
83 #endif
84 #endif
85 
86  dev = evdi_drm_device_create(&pdev->dev);
87  if (IS_ERR_OR_NULL(dev))
88  goto err_free;
89 
90  data->drm_dev = dev;
91  data->symlinked = false;
92  platform_set_drvdata(pdev, data);
93  return PTR_ERR_OR_ZERO(dev);
94 
95 err_free:
96  kfree(data);
97  return PTR_ERR_OR_ZERO(dev);
98 }

111 bool evdi_platform_device_is_free(struct platform_device *pdev)
112 {
113  struct evdi_platform_device_data *data = platform_get_drvdata(pdev);
114  struct evdi_device *evdi = data->drm_dev->dev_private;
115 
116  if (evdi && !evdi_painter_is_connected(evdi->painter) &&
117  !data->symlinked)
118  return true;
119  return false;
120 }

122 void evdi_platform_device_link(struct platform_device *pdev,
123  struct device *parent)
124 {
125  struct evdi_platform_device_data *data = NULL;
126  int ret = 0;
127 
128  if (!parent || !pdev)
129  return;
130 
131  data = platform_get_drvdata(pdev);
132  if (!evdi_platform_device_is_free(pdev)) {
133  EVDI_FATAL("Device is already attached can't symlink again\n");
134  return;
135  }
136 
137  ret = sysfs_create_link(&pdev->dev.kobj, &parent->kobj, "device");
138  if (ret) {
139  EVDI_FATAL("Failed to create sysfs link from evdi to parent device\n");
140  } else {
141  data->symlinked = true;
142  data->parent = parent;
143  }
144 }

146 void evdi_platform_device_unlink_if_linked_with(struct platform_device *pdev,
147  struct device *parent)
148 {
149  struct evdi_platform_device_data *data = platform_get_drvdata(pdev);
150 
151  if (parent && data->parent == parent) {
152  sysfs_remove_link(&pdev->dev.kobj, "device");
153  data->symlinked = false;
154  data->parent = NULL;
155  EVDI_INFO("Detached from parent device\n");
156  }
157 }
