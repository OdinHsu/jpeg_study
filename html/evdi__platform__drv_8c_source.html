43 static int evdi_platform_drv_usb(__always_unused struct notifier_block *nb,
44  unsigned long action,
45  void *data)
46 {
47  struct usb_device *usb_dev = (struct usb_device *)(data);
48  struct platform_device *pdev;
49  int i = 0;
50 
51  if (!usb_dev)
52  return 0;
53  if (action != BUS_NOTIFY_DEL_DEVICE)
54  return 0;
55 
56  for (i = 0; i < EVDI_DEVICE_COUNT_MAX; ++i) {
57  pdev = g_ctx.devices[i];
58  if (!pdev)
59  continue;
60  evdi_platform_device_unlink_if_linked_with(pdev, &usb_dev->dev);
61  if (pdev->dev.parent == &usb_dev->dev) {
62  EVDI_INFO("Parent USB removed. Removing evdi.%d\n", i);
63  evdi_platform_dev_destroy(pdev);
64  evdi_platform_drv_context_lock((&g_ctx));
65  g_ctx.dev_count--;
66  g_ctx.devices[i] = NULL;
67  evdi_platform_drv_context_unlock((&g_ctx));
68  }
69  }
70  return 0;
71 }

73 static int evdi_platform_drv_get_free_idx(struct evdi_platform_drv_context *ctx)

75  int i;

98 static struct platform_device *evdi_platform_drv_create_new_device(struct evdi_platform_drv_context *ctx)
99 {
100  struct platform_device *pdev = NULL;
101  struct platform_device_info pdevinfo = {
102  .parent = NULL,
103  .name = DRIVER_NAME,
104  .id = evdi_platform_drv_get_free_idx(ctx),
105  .res = NULL,
106  .num_res = 0,
107  .data = NULL,
108  .size_data = 0,
109  .dma_mask = DMA_BIT_MASK(32),
110  };
111 
112  if (pdevinfo.id < 0 || ctx->dev_count >= EVDI_DEVICE_COUNT_MAX) {
113  EVDI_ERROR("Evdi device add failed. Too many devices.\n");
114  return ERR_PTR(-EINVAL);
115  }
116 
117  pdev = evdi_platform_dev_create(&pdevinfo);
118  ctx->devices[pdevinfo.id] = pdev;
119  ctx->dev_count++;
120 
121  return pdev;
122 }

124 int evdi_platform_device_add(struct device *device, struct device *parent)
125 {
126  struct evdi_platform_drv_context *ctx =
127  (struct evdi_platform_drv_context *)dev_get_drvdata(device);
128  struct platform_device *pdev = NULL;
129 
130  evdi_platform_drv_context_lock(ctx);
131  if (parent)
132  pdev = evdi_platform_drv_get_free_device(ctx);
133 
134  if (IS_ERR_OR_NULL(pdev))
135  pdev = evdi_platform_drv_create_new_device(ctx);
136  evdi_platform_drv_context_unlock(ctx);
137 
138  if (IS_ERR_OR_NULL(pdev))
139  return -EINVAL;
140 
141  evdi_platform_device_link(pdev, parent);
142  return 0;
143 }

145 int evdi_platform_add_devices(struct device *device, unsigned int val)
146 {
147  unsigned int dev_count = evdi_platform_device_count(device);
148 
149  if (val == 0) {
150  EVDI_WARN("Adding 0 devices has no effect\n");
151  return 0;
152  }
153  if (val > EVDI_DEVICE_COUNT_MAX - dev_count) {
154  EVDI_ERROR("Evdi device add failed. Too many devices.\n");
155  return -EINVAL;
156  }
157 
158  EVDI_INFO("Increasing device count to %u\n", dev_count + val);
159  while (val-- && evdi_platform_device_add(device, NULL) == 0)
160  ;
161  return 0;
162 }

164 void evdi_platform_remove_all_devices(struct device *device)
165 {
166  int i;
167  struct evdi_platform_drv_context *ctx =
168  (struct evdi_platform_drv_context *)dev_get_drvdata(device);
169 
170  evdi_platform_drv_context_lock(ctx);
171  for (i = 0; i < EVDI_DEVICE_COUNT_MAX; ++i) {
172  if (ctx->devices[i]) {
173  EVDI_INFO("Removing evdi %d\n", i);
174  evdi_platform_dev_destroy(ctx->devices[i]);
175  g_ctx.dev_count--;
176  g_ctx.devices[i] = NULL;
177  }
178  }
179  ctx->dev_count = 0;
180  evdi_platform_drv_context_unlock(ctx);
181 }

207 static int __init evdi_init(void)
208 {
209  int ret;
210 
211  EVDI_INFO("Initialising logging on level %u\n", evdi_loglevel);
212  EVDI_INFO("Atomic driver: yes");
213 
214  memset(&g_ctx, 0, sizeof(g_ctx));
215  g_ctx.root_dev = root_device_register(DRIVER_NAME);
216  g_ctx.usb_notifier.notifier_call = evdi_platform_drv_usb;
217  mutex_init(&g_ctx.lock);
218  dev_set_drvdata(g_ctx.root_dev, &g_ctx);
219 
220  usb_register_notify(&g_ctx.usb_notifier);
221  evdi_sysfs_init(g_ctx.root_dev);
222  ret = platform_driver_register(&evdi_platform_driver);
223  if (ret)
224  return ret;
225 
226  if (evdi_initial_device_count)
227  return evdi_platform_add_devices(
228  g_ctx.root_dev, evdi_initial_device_count);
229 
230  return 0;
231 }

233 static void __exit evdi_exit(void)
234 {
235  EVDI_CHECKPT();
236  evdi_platform_remove_all_devices(g_ctx.root_dev);
237  platform_driver_unregister(&evdi_platform_driver);
238 
239  if (!PTR_ERR_OR_ZERO(g_ctx.root_dev)) {
240  evdi_sysfs_exit(g_ctx.root_dev);
241  usb_unregister_notify(&g_ctx.usb_notifier);
242  dev_set_drvdata(g_ctx.root_dev, NULL);
243  root_device_unregister(g_ctx.root_dev);
244  }
245  EVDI_INFO("Exit %s driver\n", DRIVER_NAME);
246 }
