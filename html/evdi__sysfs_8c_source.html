57 static ssize_t add_device_with_usb_path(struct device *dev,
58  const char *buf, size_t count)
59 {
60  char *usb_path = kstrdup(buf, GFP_KERNEL);
61  char *temp_path = usb_path;
62  char *bus_token;
63  char *usb_token;
64  char *usb_token_copy = NULL;
65  char *token;
66  char *bus;
67  char *port;
68  struct evdi_usb_addr usb_addr;
69 
70  if (!usb_path)
71  return -ENOMEM;
72 
73  memset(&usb_addr, 0, sizeof(usb_addr));
74  temp_path = strnstr(temp_path, "usb:", count);
75  if (!temp_path)
76  goto err_parse_usb_path;
77 
78 
79  temp_path = strim(temp_path);
80 
81  bus_token = strsep(&temp_path, ":");
82  if (!bus_token)
83  goto err_parse_usb_path;
84 
85  usb_token = strsep(&temp_path, ":");
86  if (!usb_token)
87  goto err_parse_usb_path;
88 
89  /* Separate trailing ':*' from usb_token */
90  strsep(&temp_path, ":");
91 
92  token = usb_token_copy = kstrdup(usb_token, GFP_KERNEL);
93  bus = strsep(&token, "-");
94  if (!bus)
95  goto err_parse_usb_path;
96  if (kstrtouint(bus, 10, &usb_addr.addr[usb_addr.len++]))
97  goto err_parse_usb_path;
98 
99  do {
100  port = strsep(&token, ".");
101  if (!port)
102  goto err_parse_usb_path;
103  if (kstrtouint(port, 10, &usb_addr.addr[usb_addr.len++]))
104  goto err_parse_usb_path;
105  } while (token && port && usb_addr.len < MAX_EVDI_USB_ADDR);
106 
107  if (evdi_platform_device_attach(dev, &usb_addr) != 0) {
108  EVDI_ERROR("Unable to attach to: %s\n", buf);
109  kfree(usb_path);
110  kfree(usb_token_copy);
111  return -EINVAL;
112  }
113 
114  EVDI_INFO("Attaching to %s:%s\n", bus_token, usb_token);
115  kfree(usb_path);
116  kfree(usb_token_copy);
117  return count;
118 
119 err_parse_usb_path:
120  EVDI_ERROR("Unable to parse usb path: %s", buf);
121  kfree(usb_path);
122  kfree(usb_token_copy);
123  return -EINVAL;
124 }

126 static int find_usb_device_at_path(struct usb_device *usb, void *data)
127 {
128  struct evdi_usb_addr *find_path = (struct evdi_usb_addr *)(data);
129  struct usb_device *pdev = usb;
130  int port = 0;
131  int i;
132 
133  i = find_path->len - 1;
134  while (pdev != NULL && i >= 0 && i < MAX_EVDI_USB_ADDR) {
135  port = pdev->portnum;
136  if (port == 0)
137  port = pdev->bus->busnum;
138 
139  if (port != find_path->addr[i])
140  return 0;
141 
142  if (pdev->parent == NULL && i == 0) {
143  find_path->usb = usb;
144  return 1;
145  }
146  pdev = pdev->parent;
147  i--;
148  }
149 
150  return 0;
151 }

153 static int evdi_platform_device_attach(struct device *device,
154  struct evdi_usb_addr *parent_addr)
155 {
156  struct device *parent = NULL;
157 
158  if (!parent_addr)
159  return -EINVAL;
160 
161  if (!usb_for_each_dev(parent_addr, find_usb_device_at_path) ||
162  !parent_addr->usb)
163  return -EINVAL;
164 
165  parent = &parent_addr->usb->dev;
166  return evdi_platform_device_add(device, parent);
167 }
