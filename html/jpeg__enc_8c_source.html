319 void writebits(bitstring bs,cinfo_t *cinfo)
320 {
321  WORD value;
322  int i,len,shift;
323 #ifdef _X86_64_
324  INT64 data;
325 #else
326  DWORD data;
327 #endif
328  BYTE* p;
329 
330  len = bs & 0xFF; // length;
331  value = (WORD)(bs>>len); // left alignment to WORD
332 
333  if( (cinfo->wordpos + len) < BUF_BITS ) { // 32/64
334  cinfo->wordnew = (cinfo->wordnew << len) | (value >> (16-len));
335  cinfo->wordpos += len;
336  } else { // >= 32/64
337  shift = BUF_BITS - cinfo->wordpos;
338  data = (cinfo->wordnew << shift) | (value >> (16-shift));
339 
340  // output 4/8 bytes
341  p = ((BYTE*)&data)+(BUF_BYTES-1);
342  for(i=0;i<BUF_BYTES;i++) {
343  writebyte(&cinfo->jpeg_stream,*p);
344  if( *p-- == 0xFF ) writebyte(&cinfo->jpeg_stream,0x00);
345  }
346 
347  value <<= shift;
348  len -= shift;
349 
350  cinfo->wordnew = value >> (16-len);
351  cinfo->wordpos = len;
352  }
353 };

387 void compute_Huffman_table(BYTE *nrcodes,BYTE *std_table,bitstring *HT)
388 {
389  BYTE k,j;
390  BYTE pos_in_table;
391  WORD codevalue;
392  codevalue=0; pos_in_table=0;
393  for (k=1;k<=16;k++) {
394  for (j=1;j<=nrcodes[k];j++) {
395  HT[std_table[pos_in_table]] = (codevalue<<16) | k;
396  pos_in_table++;
397  codevalue++;
398  }
399  codevalue*=2;
400  }
401 }

426 static int compute_reciprocal (UINT16 divisor, DCTELEM * dtbl)
427 {
428  UDCTELEM2 fq, fr;
429  UDCTELEM c;
430  int b, r;
431 
432  b = flss(divisor) - 1;
433  r = sizeof(DCTELEM) * 8 + b;
434 
435  fq = ((UDCTELEM2)1 << r) / divisor;
436  fr = ((UDCTELEM2)1 << r) % divisor;
437 
438  c = divisor / 2; /* for rounding */
439 
440  if (fr == 0) { /* divisor is power of two */
441  /* fq will be one bit too large to fit in DCTELEM, so adjust */
442  fq >>= 1;
443  r--;
444  } else if (fr <= (divisor / 2U)) { /* fractional part is < 0.5 */
445  c++;
446  } else { /* fractional part is > 0.5 */
447  fq++;
448  }
449 
450  dtbl[DCTSIZE2 * 0] = (DCTELEM) fq; /* reciprocal */
451  dtbl[DCTSIZE2 * 1] = (DCTELEM) c; /* correction + roundfactor */
452  dtbl[DCTSIZE2 * 2] = (DCTELEM) (1 << (sizeof(DCTELEM)*8*2 - r)); /* scale */
453  dtbl[DCTSIZE2 * 3] = (DCTELEM) r - sizeof(DCTELEM)*8; /* shift */
454 
455  if(r <= 16) return 0;
456  else return 1;
457 }

505 #ifndef _PROCESS_DU_
506 void process_DU(cinfo_t *cinfo,INT16 *DU,SWORD *DC,HuffmanTable_t* HT)
507 {
508  int i;
509  int startpos;
510  int end0pos;
511  int nrzeroes;
512  int Diff;
513  int mask;
514  int value;
515  int nrmarker;
516  bitstring *HTDC=HT->DC;
517  bitstring *HTAC=HTDC+12;
518 
519  DWORD cat;
520  bitstring data;
521 
522  bitstring EOB=HTAC[0x00];
523  bitstring M16zeroes=HTAC[0xF0];
524 
525  INT16 ZDU[64];
526 
527  //zigzag reorder
528  for (i=0;i<=63;i++) ZDU[zigzag[i]]=DU[i];
529  Diff=DU[0]-*DC;
530  *DC=DU[0];
531 
532  //Encode DC
533 
534  if (Diff==0) {
535  writebits(HTDC[0],cinfo); //Diff might be 0
536  } else {
537 
538  mask = (Diff>>31); // -1 or 0
539  value = Diff + mask;
540 
541  cat = (BYTE)(32 - clz(value^mask));
542  writebits(HTDC[cat],cinfo);
543  writebits((value << 16) | (cat),cinfo);
544 
545  }
546 
547  //Encode ACs
548  for (end0pos=63;(end0pos>0)&&(ZDU[end0pos]==0);end0pos--) ;
549  //end0pos = first element in reverse order !=0
550  if (end0pos==0) {
551  writebits(EOB,cinfo);
552  return;
553  }
554 
555  i=1;
556  while (i<=end0pos) {
557  startpos=i;
558  for (; (ZDU[i]==0)&&(i<=end0pos);i++) ;
559  nrzeroes=i-startpos;
560  if (nrzeroes>=16) {
561  for (nrmarker=0;nrmarker<nrzeroes/16;nrmarker++) writebits(M16zeroes,cinfo);
562  nrzeroes=nrzeroes%16;
563  };
564  mask = ((int)ZDU[i])>>31; // -1 or 0
565  value = ZDU[i] + mask;
566  cat = (BYTE)(32 - clz(value^mask));
567  data = (value<<16)|cat;
568  writebits(HTAC[nrzeroes*16+cat],cinfo);
569  writebits(data,cinfo);
570 
571  i++;
572  }
573 
574  if (end0pos!=63) writebits(EOB,cinfo);
575 
576 }
