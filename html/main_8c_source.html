130  vendor_id = 0x34c7;
131  product_id = 0x2104;
132  class_id = LIBUSB_HOTPLUG_MATCH_ANY;
133 
134 
135 
136 //============================== libusb hotplug ===============================
137  rc = libusb_init (NULL);
138  if (rc < 0)
139  {
140  printf("failed to initialise libusb: %s\n", libusb_error_name(rc));
141  return EXIT_FAILURE;
142  }
143 
144  if (!libusb_has_capability (LIBUSB_CAP_HAS_HOTPLUG)) {
145  printf ("Hotplug capabilites are not supported on this platform\n");
146  libusb_exit (NULL);
147  return EXIT_FAILURE;
148  }
149 
150  rc = libusb_hotplug_register_callback (NULL, LIBUSB_HOTPLUG_EVENT_DEVICE_ARRIVED, LIBUSB_HOTPLUG_NO_FLAGS, vendor_id,
151  product_id, class_id, hotplug_callback, NULL, &hp[0]);
152  if (LIBUSB_SUCCESS != rc) {
153  fprintf (stderr, "Error registering callback 0\n");
154  libusb_exit (NULL);
155  return EXIT_FAILURE;
156  }
157 
158  rc = libusb_hotplug_register_callback (NULL, LIBUSB_HOTPLUG_EVENT_DEVICE_LEFT, LIBUSB_HOTPLUG_NO_FLAGS, vendor_id,
159  product_id,class_id, hotplug_callback_detach, NULL, &hp[1]);
160  if (LIBUSB_SUCCESS != rc) {
161  fprintf (stderr, "Error registering callback 1\n");
162  libusb_exit (NULL);
163  return EXIT_FAILURE;
164  }
165  return 0;
166
