332 /* v refresh frequency in Hz */
333 static double
334 mode_refresh (const XRRModeInfo *mode_info)
335 {
336  double rate;
337  double vTotal = mode_info->vTotal;
338 
339  if (mode_info->modeFlags & RR_DoubleScan) {
340  /* doublescan doubles the number of lines */
341  vTotal *= 2;
342  }
343 
344  if (mode_info->modeFlags & RR_Interlace) {
345  /* interlace splits the frame into two fields */
346  /* the field rate is what is typically reported by monitors */
347  vTotal /= 2;
348  }
349 
350  if (mode_info->hTotal && vTotal)
351  rate = ((double) mode_info->dotClock /
352  ((double) mode_info->hTotal * (double) vTotal));
353  else
354  rate = 0;
355  return rate;
356 }

510 static XRRModeInfo *
511 find_mode (name_t *name, double refresh)
512 {
513  int m;
514  XRRModeInfo *best = NULL;
515  double bestDist = 0;
516 
517  for (m = 0; m < res->nmode; m++)
518  {
519  XRRModeInfo *mode = &res->modes[m];
520  if ((name->kind & name_xid) && name->xid == mode->id)
521  {
522  best = mode;
523  break;
524  }
525  if ((name->kind & name_string) && !strcmp (name->string, mode->name))
526  {
527  double dist;
528 
529  if (refresh)
530  dist = fabs (mode_refresh (mode) - refresh);
531  else
532  dist = 0;
533  if (!best || dist < bestDist)
534  {
535  bestDist = dist;
536  best = mode;
537  }
538  }
539  }
540  return best;
541 }

553 static
554 XRRModeInfo *
555 find_mode_for_output (output_t *output, name_t *name)
556 {
557  XRROutputInfo *output_info = output->output_info;
558  int m;
559  XRRModeInfo *best = NULL;
560  double bestDist = 0;
561 
562  for (m = 0; m < output_info->nmode; m++)
563  {
564  XRRModeInfo *mode;
565 
566  mode = find_mode_by_xid (output_info->modes[m]);
567  if (!mode) continue;
568  if ((name->kind & name_xid) && name->xid == mode->id)
569  {
570  best = mode;
571  break;
572  }
573  if ((name->kind & name_string) && !strcmp (name->string, mode->name))
574  {
575  double dist;
576 
577  /* Stay away from doublescan modes unless refresh rate is specified. */
578  if (!output->refresh && (mode->modeFlags & RR_DoubleScan))
579  continue;
580 
581  if (output->refresh)
582  dist = fabs (mode_refresh (mode) - output->refresh);
583  else
584  dist = 0;
585  if (!best || dist < bestDist)
586  {
587  bestDist = dist;
588  best = mode;
589  }
590  }
591  }
592  return best;
593 }

595 static XRRModeInfo *
596 preferred_mode (output_t *output)
597 {
598  XRROutputInfo *output_info = output->output_info;
599  int m;
600  XRRModeInfo *best;
601  int bestDist;
602 
603  best = NULL;
604  bestDist = 0;

606  {
607  XRRModeInfo *mode_info = find_mode_by_xid (output_info->modes[m]);
608  int dist;
609 
610  if (m < output_info->npreferred)
611  dist = 0;
612  else if (output_info->mm_height)
613  dist = (1000 * DisplayHeight(dpy, screen) / DisplayHeightMM(dpy, screen) -
614  1000 * mode_info->height / output_info->mm_height);
615  else
616  dist = DisplayHeight(dpy, screen) - mode_info->height;
617 
618  if (dist < 0) dist = -dist;
619  if (!best || dist < bestDist)
620  {
621  best = mode_info;
622  bestDist = dist;
623  }
624  }
